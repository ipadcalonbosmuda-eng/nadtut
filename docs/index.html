Instalasi (contoh)

# 1) Install SDK (placeholder)
npm i @plasmatic/sdk

# 2) Inisialisasi
import { TokenFactory } from "@plasmatic/sdk"

const tf = new TokenFactory(provider, signer)
const tx = await tf.create({
  name: "PLAS",
  symbol: "PLS",
  supply: "1_000_000",
})

# 3) Verifikasi di explorer (alternatif selain Sourcify)
await tf.verifyOnExplorer(tx.contract)
